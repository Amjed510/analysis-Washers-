import 'package:flutter/services.dart';

class SunmiPaymentChannel {
  static const _channel = MethodChannel('com.washpos/payment');
  
  static Future<Map<String, dynamic>> processNFCPayment({
    required double amount,
    required String invoiceId,
  }) async {
    try {
      final result = await _channel.invokeMethod('processNFCPayment', {
        'amount': (amount * 100).round(), // Halalas
        'invoiceId': invoiceId,
        'currencyCode': 'SAR',
      });
      return Map<String, dynamic>.from(result);
    } on PlatformException catch (e) {
      return {'success': false, 'error': e.message};
    }
  }
  
  static Future<void> cancelPayment() async {
    await _channel.invokeMethod('cancelPayment');
  }
}

class SyncService extends GetxService {
  final _isar = Get.find<Isar>();
  final _api = Get.find<ApiProvider>();
  final _connectivity = Get.find<ConnectivityService>();
  
  @override
  void onInit() {
    super.onInit();
    // Ù…Ø±Ø§Ù‚Ø¨Ø© Ø§Ù„Ø§ØªØµØ§Ù„
    ever(_connectivity.isConnected, (connected) {
      if (connected) syncPending();
    });
  }
  
  Future<void> syncPending() async {
    final pending = await _isar.localTransactions
        .filter()
        .syncStatusEqualTo(SyncStatus.pending)
        .findAll();
    
    for (final tx in pending) {
      await _syncTransaction(tx);
    }
  }
  
  Future<void> _syncTransaction(LocalTransaction tx) async {
    try {
      await _isar.writeTxn(() async {
        tx.syncStatus = SyncStatus.syncing;
        await _isar.localTransactions.put(tx);
      });
      
      final response = await _api.syncTransaction(
        data: tx.invoiceData,
        idempotencyKey: tx.localId,
      );
      
      await _isar.writeTxn(() async {
        tx.serverId = response.id;
        tx.syncStatus = SyncStatus.synced;
        tx.syncedAt = DateTime.now();
        await _isar.localTransactions.put(tx);
      });
      
    } catch (e) {
      await _isar.writeTxn(() async {
        tx.retryCount++;
        tx.syncStatus = tx.retryCount >= 5 
            ? SyncStatus.failed 
            : SyncStatus.pending;
        await _isar.localTransactions.put(tx);
      });
    }
  }
}
